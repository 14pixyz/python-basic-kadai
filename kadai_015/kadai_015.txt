class Human:
    def __init__(self):
        self.name = ""
        self.age = ""

    # メソッドを定義する
    def printinfo(self, name, age):
        self.name = name
        self.age = age
        print("名前:" + self.name + "  " + "年齢:" + str(self.age))

# インスタンス化する
human = Human()

human.printinfo("太郎", 30)
